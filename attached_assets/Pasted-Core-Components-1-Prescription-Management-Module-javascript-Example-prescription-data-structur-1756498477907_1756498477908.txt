// Express.js API routes
app.post('/api/prescriptions/send', async (req, res) => {
    try {
        const prescription = req.body;
        const result = await pharmacyService.sendPrescription(prescription);
        res.json({ success: true, trackingId: result.trackingId });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

app.get('/api/pharmacies/search', async (req, res) => {
    const { zipCode, radius, insurance } = req.query;
    const pharmacies = await pharmacyService.findNearbyPharmacies(
        zipCode, radius, insurance
    );
    res.json(pharmacies);
});

// React component example
const PharmacySelector = ({ patientLocation, insurance }) => {
    const [pharmacies, setPharmacies] = useState([]);
    const [selectedPharmacy, setSelectedPharmacy] = useState(null);
    
    useEffect(() => {
        fetchNearbyPharmacies(patientLocation, insurance)
            .then(setPharmacies);
    }, [patientLocation, insurance]);
    
    return (
        <div className="pharmacy-selector">
            <h3>Select Pharmacy</h3>
            {pharmacies.map(pharmacy => (
                <PharmacyCard 
                    key={pharmacy.id}
                    pharmacy={pharmacy}
                    onSelect={setSelectedPharmacy}
                />
            ))}
        </div>
    );
};
